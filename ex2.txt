class owner
{
    String name;
    String city;
    String profession;
    owner(String na,String ci,String pro )
    {
       name=na;
       city=ci;
       profession=pro; 
    }
    String getname()
    {
        return name;
    }
    String getcity()
    {
        return city;
    }
    
     String getprofession()
    {
        return profession;
    }
    @Override
    public String toString()
    {
        return "name :"+name+" city: "+city+" profession "+profession;
    }

}
